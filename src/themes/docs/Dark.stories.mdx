import { Meta } from '@storybook/addon-docs/blocks';
import { ColorPalette, ColorItem } from '@storybook/components';

import { dark } from '../.';

import tokens from '../../tokens';
import { FigmaImage } from '../../docs';

<Meta title="Themes/Dark" />

export function getTokenNameByValue(value) {
	const designToken = Object.entries(tokens.colors).find(([k, v]) => v === value);
	return designToken ? designToken[0] : 'there is no design token used here!';
}
