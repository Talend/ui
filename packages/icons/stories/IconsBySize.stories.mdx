import { Meta, IconGallery } from '@storybook/addon-docs';

import Icon, { realIconSizes, getTShirtSize, IconToolbar, IconItem } from './Icon';
import { infoFromFigma as icons } from '../dist/info';

<Meta title="Icons by size" />

# Icons

<IconToolbar />

<div>
	{realIconSizes.map((realIconSize, i) => (
		<section key={i}>
			<h2>{getTShirtSize(realIconSize)}</h2>
			<IconGallery>
				{Object.entries(icons)
					.filter(([, size]) => size === realIconSize)
					.map(([name, size], j) => {
						const currentTShirtSize = getTShirtSize(size);
						return (
							<IconItem key={j} name={name.split(':')[0] + ':' + currentTShirtSize}>
								<Icon size={currentTShirtSize} name={name} />
							</IconItem>
						);
					})}
			</IconGallery>
		</section>
	))}
</div>
