import { Meta, IconGallery } from '@storybook/addon-docs';

import Icon, { realIconSizes, IconToolbar, IconItem } from './Icon';
import { infoFromFigma as icons } from '../dist/info';

<Meta title="Icons by size" />

# Icons

<IconToolbar />

This page helps you figure out how many icons are available per size.

<div>
	{realIconSizes.map((realIconSize, i) => (
		<section key={i}>
			<h2>{realIconSize}</h2>
			<IconGallery>
				{Object.entries(icons)
					.filter(([, size]) => size === realIconSize)
					.map(([name, size], j) => {
						return (
							<IconItem key={j} name={name.split(':')[0] + ':' + size}>
								<Icon size={size} name={name} />
							</IconItem>
						);
					})}
			</IconGallery>
		</section>
	))}
</div>
